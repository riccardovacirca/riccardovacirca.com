Skeleton di una applicazione APR

#include "apr.h"
#include "apr_pools.h"
#include "stdio.h"

int main(int argc, char **argv) {

  int res = 0;

  struct state_t {
    int init, pool;
  } state = {0, 0}

  do {

    apr_status_t rv;
    
    rv = apr_initialize();
    state.init = rv == APR_SUCCESS;
    if (!state.init)
      break;
    
    apr_pool_t *mp;
    rv = apr_pool_create(&mp, NULL);
    state.pool = rv == APR_SUCCESS;
    if (!state.pool)
      break;

    res = 1;
  
    apr_pool_destroy(mp);
    apr_terminate();
  
  } while (0);

  if (!res) {
    if (!state.init) {
      printf("APR initialization error\n");
    } else if (!state.pool) {
      printf("APR memory error\n");
    }
  }

  exit(res ? EXIT_FAILURE : EXIT_SUCCESS);
}

Funzione della logica applicativa

#include "apr.h"
#include "apr_pools.h"
#include "stdio.h"

int example_logic(apr_pool_t *mp) {

  int res = 0;

  struct state_t {
    int input;
  } state = {0}

  do {

    state.input = mp != NULL;
    if (!state.input)
      break;

    printf("Hello, World!\n");
      
    res = 1;
    
  } while (0);

  if (!res) {
    if (!state.input) {
      printf("Input error\n");
    }
  }

  return res;
}
