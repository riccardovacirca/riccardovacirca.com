Main function

// Copyright (C)2023-2024 - Riccardo Vacirca

#include "apr.h"
#include "apr_pools.h"
#include "stdio.h"

int main(int argc, char **argv) {

  int res = 1;
  apr_status_t rv;
  apr_pool_t *mp;

  struct state_t {
    int init, pool;
  } st = {0, 0}

  do {
    
    rv = apr_initialize();
    st.init = rv == APR_SUCCESS;
    if (!st.init) {
      break;
    }
    
    rv = apr_pool_create(&mp, NULL);
    st.pool = rv == APR_SUCCESS;
    if (!st.pool) {
      break;
    }

    res = 0;

  } while (0);

  if (res) {
    if (!st.init) {
      printf("APR initialization error\n");
    } else if (!st.pool) {
      printf("APR memory error\n");
    } else {
      printf("General error\n");
    }
  }

  if (st.init) {
    if (st.pool) {
      apr_pool_destroy(mp);
    }
    apr_terminate();
  }

  return res;
}

Application logic functions

// Copyright (C)2023-2024 - Riccardo Vacirca

#include "apr.h"
#include "apr_pools.h"
#include "stdio.h"

int example_logic(apr_pool_t *mp) {

  int res = 0;

  struct state_t {
    int input;
  } st = {0}

  do {

    st.input = mp != NULL;
    if (!st.input) {
      break;
    }

    printf("Hello, World!\n");
      
    res = 1;
    
  } while (0);

  if (!res) {
    if (!st.input) {
      printf("Input error\n");
    } else {
      printf("General error\n");
    }
  }

  return res;
}
